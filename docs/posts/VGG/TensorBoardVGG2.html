class CustomDataset(torch.utils.data.Dataset):
    def __init__(self, root_dir, transform = None):
        self.root_dir = root_dir
        self.transform = transform
        self.images = []
        self.labels = []

        for img_name in os.listdir(root_dir):
            img_path = os.path.join(root_dir, img_name)
            if os.path.isfile(img_path):
                if img_name.startswith("Gulab_Jamun"):
                    label_index = 0
                elif img_name.startswith("Jalebi"):
                    label_index = 1
                else:
                    continue
                self.images.append(img_path)
                self.labels.append(label_index)

    def __len__(self):
        return len(self.images)

    def __getitem__(self, idx):
        img_path = self.images[idx]
        label = self.labels[idx]
        image = Image.open(img_path).convert("RGB")

        if self.transform:
            image = self.transform(image)

        return image, label

def plot_images(dataset_loader, dataset_name, num_images = 5, label_mapping = None):
    num_rows = num_images // 5

    fig, axes = plt.subplots(num_rows, 5, figsize = (15, 3 * num_rows))
    fig.suptitle(f"Sample images from {dataset_name}", fontsize = 16)

    for batch_idx, (images, labels) in enumerate(dataset_loader):
        images = (images + 1) / 2

        for i in range(images.size(0)):
            row_idx = (batch_idx * 5 + i) // 5
            col_idx = (batch_idx * 5 + i) % 5
            if (batch_idx * 5 + i) >= num_images:
                break
            if num_rows == 1:
                ax = axes[col_idx]
            else:
                ax = axes[row_idx, col_idx]

            rgb_images = images[i].permute(1, 2, 0)
            ax.imshow(rgb_images)

            if label_mapping is not None:
                ax.set_title(f"{label_mapping[labels[i].item()]} ({labels[i].item()})")
            else:
                ax.set_title(f"{labels[i].item()}")

        if (batch_idx + 1) * len(images) >= num_images:
            break

    plt.show()

class VGG1(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(3, 64, kernel_size = 3, padding = 1)
        self.pool1 = nn.MaxPool2d(kernel_size = 2, stride = 2)
        self.lin1 = nn.Linear(64 * 112 * 112, 256)
        self.lin2 = nn.Linear(256, 1)
        self.relu = nn.ReLU()

    def forward(self, x):
        x = self.relu(self.conv1(x))
        x = self.pool1(x)
        x = x.view(-1, 64 * 112 * 112)
        x = self.relu(self.lin1(x))
        logits = self.lin2(x)
        return logits

Template code to send images to Tensorboard

def plot_images_with_predictions(model, dataset, label_mapping = None, returnFig = False):
    num_images = len(dataset)
    num_rows = (num_images // 8) + 1
    total_correct = 0
    fig, axes = plt.subplots(num_rows, 8, figsize=(20, 3 * num_rows + 3))
    model.eval()

    with torch.no_grad():
        images_list, labels_list = [entry[0] for entry in dataset], [entry[1] for entry in dataset]
        images, labels = torch.stack(images_list), torch.tensor(labels_list)
        images, labels = images.to(device), labels.to(device)
        logits = model(images)
        predictions = (logits > 0)

        images = (images + 1) / 2

        for i in range(images.size(0)):
            image_np = images[i].cpu().permute(1, 2, 0).numpy()
            true_label = labels[i].item()
            predicted_label = predictions[i].item()

            if true_label == predicted_label:
                total_correct += 1

            if label_mapping is not None:
                true_labelN = label_mapping[true_label]
                predicted_labelN = label_mapping[predicted_label]

            row_idx = i // 8
            col_idx = i % 8
            ax = axes[row_idx, col_idx] if num_rows > 1 else axes[col_idx]

            title_color = "green" if true_label == predicted_label else "red"
            ax.imshow(image_np)
            ax.set_title(f"T: {true_labelN} P: {predicted_labelN}", color=title_color, fontsize=12)
            ax.axis("off")

    fig.suptitle(f"Accuracy: {total_correct}/{images.size(0)}" "\n" f"{100 * (total_correct / images.size(0))}", fontsize=30, color="blue")
    plt.tight_layout(rect=[0, 0.03, 1, 0.95])

    if returnFig:
        return fig
    else:
        plt.show()

Functions for Training and Final Prediction

def trainDataset(model, train_loader, test_loader, test_dataset, num_epochs = 10, lr = 0.01, fileName = None):
    criterion = nn.BCEWithLogitsLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr = lr)

    writer = SummaryWriter(f"./{fileName}")
    total_steps = 0
    start_time = time.time()

    for epoch in range(num_epochs):
        model.train()
        running_loss = 0.0
        correct_train = 0
        total_train = 0

        for batch_idx, (images, labels) in enumerate(train_loader):
            images, labels = images.to(device), labels.to(device)

            optimizer.zero_grad()
            logits = model(images)
            loss = criterion(logits, labels.float().view(-1, 1))
            loss.backward()
            optimizer.step()

            running_loss += loss.item()
            total_steps += 1

            preds = logits > 0
            correct_train += (preds == labels.view_as(preds)).sum().item()
            total_train += labels.size(0)

            writer.add_scalar("Training Loss", running_loss / total_steps, global_step = total_steps)

            train_acc = 100 * correct_train / total_train
            writer.add_scalar("Training Accuracy", train_acc, global_step=total_steps)

            model.eval()
            correct_test = 0
            total_test = 0

            with torch.no_grad():
                for images, labels in test_loader:
                    images, labels = images.to(device), labels.to(device)
                    logits = model(images)
                    preds = logits > 0
                    correct_test += (preds == labels.view_as(preds)).sum().item()
                    total_test += labels.size(0)

            test_acc = 100 * correct_test / total_test
            writer.add_scalar("Testing Accuracy", test_acc, global_step=total_steps)

            if (batch_idx + 1) % 10 == 0:
                print(f"Epoch: {epoch + 1}/{num_epochs} | Step: {total_steps} | Training Loss: {running_loss / total_steps} | Training Accuracy: {train_acc} | Testing Accuracy: {test_acc}")
                fig = plot_images_with_predictions(model, test_dataset, my_labels, returnFig = True)
                writer.add_figure("Predictions", fig, global_step = total_steps)

    end_time = time.time()
    training_time = end_time - start_time
    num_params = sum(p.numel() for p in model.parameters() if p.requires_grad)

    writer.close()

    return {
        "Training Time": training_time,
        "Training Loss": running_loss / total_steps,
        "Training Accuracy": train_acc,
        "Testing Accuracy": test_acc,
        "Number of Model Parameters": num_params
    }


def finalPrediction(model, test_dataset):
    model.eval()
    correct_test = 0
    total_test = 0
    with torch.no_grad():
        images_list, labels_list = [entry[0] for entry in test_dataset], [entry[1] for entry in test_dataset]
        images, labels = torch.stack(images_list), torch.tensor(labels_list)
        images, labels = images.to(device), labels.to(device)
        logits = model(images)
        preds = logits > 0
        correct_test += (preds == labels.view_as(preds)).sum().item()
        total_test += labels.size(0)

    test_acc = 100 * correct_test / total_test
    print(f"Final Testing Accuracy: {test_acc}")

\[ \texttt{VGG3 Architecture} \]

class VGG3(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(3, 64, kernel_size = 3, padding = 1)
        self.pool1 = nn.MaxPool2d(kernel_size = 2, stride = 2)

        self.conv2 = nn.Conv2d(64, 128, kernel_size = 3, padding = 1)
        self.pool2 = nn.MaxPool2d(kernel_size = 2, stride = 2)

        self.conv3 = nn.Conv2d(128, 256, kernel_size = 3, padding = 1)
        self.pool3 = nn.MaxPool2d(kernel_size = 2, stride = 2)

        self.lin1 = nn.Linear(256 * 28 * 28, 512)
        self.lin2 = nn.Linear(512, 256)
        self.lin3 = nn.Linear(256, 1)

        self.relu = nn.ReLU()

    def forward(self, x):
        x = self.relu(self.conv1(x))
        x = self.pool1(x)
        x = self.relu(self.conv2(x))
        x = self.pool2(x)
        x = self.relu(self.conv3(x))
        x = self.pool3(x)
        x = x.view(-1, 256 * 28 * 28)
        x = self.relu(self.lin1(x))
        x = self.relu(self.lin2(x))
        logits = self.lin3(x)
        return logits
